# Uygulama ikonu için PNG dosyanızı aşağıdaki tüm mipmap klasörlerine kopyalayın ve adını ic_launcher.png yapın.
# Örnek: android/app/src/main/res/mipmap-xxxhdpi/ic_launcher.png
# PNG dosyanızın adı: ic_launcher.png
# Boyutlar:
# - xxxhdpi: 192x192
# - xxhdpi: 144x144
# - xhdpi: 96x96
# - hdpi: 72x72
# - mdpi: 48x48
#
# PNG dosyanızı bu klasörlere manuel olarak ekleyin.
